name: 004 Multi-Env CI/CD

on:
  pull_request:
    branches: [main]
  workflow_dispatch:
    inputs:
      environment:
        description: "Target environment"
        required: true
        default: "dev"
        type: choice
        options:
          - dev
          - qa
          - prod

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

env:
  APP_NAME: ${{ vars.APP_NAME }}

jobs:
  # ── Validate ─────────────────────────────────────────────────
  validate:
    name: Bicep Validate
    runs-on: ubuntu-latest
    environment: dev
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Azure CLI Login
        uses: azure/login@v1
        with:
          creds: ${{ secrets.AZURE_CREDENTIALS }}

      - name: Validate Bicep (dev)
        run: |
          az deployment group validate \
            --resource-group ${{ vars.AZURE_RESOURCE_GROUP }} \
            --template-file bicep/main.bicep \
            --parameters bicep/parameters.dev.json \
            --parameters appName=${{ env.APP_NAME }} environment=dev

      - name: Validate Bicep (qa)
        run: |
          az deployment group validate \
            --resource-group rg-aigenius4-qa \
            --template-file bicep/main.bicep \
            --parameters bicep/parameters.qa.json \
            --parameters appName=${{ env.APP_NAME }} environment=qa

      - name: Validate Bicep (prod)
        run: |
          az deployment group validate \
            --resource-group rg-aigenius4-prod \
            --template-file bicep/main.bicep \
            --parameters bicep/parameters.prod.json \
            --parameters appName=${{ env.APP_NAME }} environment=prod

  # ── Dev ───────────────────────────────────────────────────────
  plan-dev:
    name: Bicep Plan (dev)
    runs-on: ubuntu-latest
    needs: validate
    environment: dev
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Azure CLI Login
        uses: azure/login@v1
        with:
          creds: ${{ secrets.AZURE_CREDENTIALS }}

      - name: What-if plan (dev)
        run: |
          az deployment group what-if \
            --resource-group ${{ vars.AZURE_RESOURCE_GROUP }} \
            --template-file bicep/main.bicep \
            --parameters bicep/parameters.dev.json \
            --parameters appName=${{ env.APP_NAME }} environment=dev

  deploy-dev:
    name: Deploy (dev)
    runs-on: ubuntu-latest
    needs: plan-dev
    environment: dev
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Azure CLI Login
        uses: azure/login@v1
        with:
          creds: ${{ secrets.AZURE_CREDENTIALS }}

      - name: Deploy Bicep (dev)
        run: |
          az deployment group create \
            --resource-group ${{ vars.AZURE_RESOURCE_GROUP }} \
            --template-file bicep/main.bicep \
            --parameters bicep/parameters.dev.json \
            --parameters appName=${{ env.APP_NAME }} environment=dev

  # ── QA ────────────────────────────────────────────────────────
  plan-qa:
    name: Bicep Plan (qa)
    runs-on: ubuntu-latest
    needs: deploy-dev
    environment: qa
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Azure CLI Login
        uses: azure/login@v1
        with:
          creds: ${{ secrets.AZURE_CREDENTIALS }}

      - name: What-if plan (qa)
        run: |
          az deployment group what-if \
            --resource-group ${{ vars.AZURE_RESOURCE_GROUP }} \
            --template-file bicep/main.bicep \
            --parameters bicep/parameters.qa.json \
            --parameters appName=${{ env.APP_NAME }} environment=qa

  deploy-qa:
    name: Deploy (qa)
    runs-on: ubuntu-latest
    needs: plan-qa
    environment: qa
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Azure CLI Login
        uses: azure/login@v1
        with:
          creds: ${{ secrets.AZURE_CREDENTIALS }}

      - name: Deploy Bicep (qa)
        run: |
          az deployment group create \
            --resource-group ${{ vars.AZURE_RESOURCE_GROUP }} \
            --template-file bicep/main.bicep \
            --parameters bicep/parameters.qa.json \
            --parameters appName=${{ env.APP_NAME }} environment=qa

  # ── Prod ──────────────────────────────────────────────────────
  plan-prod:
    name: Bicep Plan (prod)
    runs-on: ubuntu-latest
    needs: deploy-qa
    environment: prod
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Azure CLI Login
        uses: azure/login@v1
        with:
          creds: ${{ secrets.AZURE_CREDENTIALS }}

      - name: What-if plan (prod)
        run: |
          az deployment group what-if \
            --resource-group ${{ vars.AZURE_RESOURCE_GROUP }} \
            --template-file bicep/main.bicep \
            --parameters bicep/parameters.prod.json \
            --parameters appName=${{ env.APP_NAME }} environment=prod

  deploy-prod:
    name: Deploy (prod)
    runs-on: ubuntu-latest
    needs: plan-prod
    environment: prod
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Azure CLI Login
        uses: azure/login@v1
        with:
          creds: ${{ secrets.AZURE_CREDENTIALS }}

      - name: Deploy Bicep (prod)
        run: |
          az deployment group create \
            --resource-group ${{ vars.AZURE_RESOURCE_GROUP }} \
            --template-file bicep/main.bicep \
            --parameters bicep/parameters.prod.json \
            --parameters appName=${{ env.APP_NAME }} environment=prod
